whales <- read.csv("~/Desktop/CSUMB/25-26/Spring 26/Stats 210 /Labs/Right Whales/stats210_portfolio_RW/23305_RWSAS.csv")
whales_clean <- whales |> 
  select(-DUPLICATE, -MOM_CALF) |> 
  mutate(sighting = dmy(SIGHTINGDATE),
         certainty = as.factor(CERTAINTY),
         category = as.factor(CATEGORY)) |> 
  select(-SIGHTINGDATE, -CERTAINTY, -CATEGORY) |> 
  drop_na() |> 
   arrange(sighting) |> 
  filter(certainty == "Definite") |> 
  mutate(
    area = case_when(
      LAT > 40 ~ "North",
      LAT > 35 ~ "Mid",
      TRUE ~ "South")) 
pivot_whale <- whales_clean |> 
  group_by(category) |> 
  summarise(mean_group = mean(GROUPSIZE, na.rm = TRUE),
            count = n())

mom_baby <- whales |> 
  select(-DUPLICATE, -CATEGORY) |> 
  mutate(
    SIGHTINGDATE = dmy(SIGHTINGDATE),
    month = month(SIGHTINGDATE, label = TRUE),
    season = case_when(
      month %in% c("Mar","Apr","May") ~ "spring",
      month %in% c("Jun", "Jul", "Aug") ~ "summer",
      month %in% c("Sep", "Oct", "Nov") ~ "fall",
      month %in%  c("Dec", "Jan", "Feb") ~ "winter"
  )) |> 
  mutate(
    area = case_when(
      LAT > 40 ~ "North",
      LAT > 35 ~ "Mid",
      TRUE ~ "South"
    )
  )


baby_area_count <- mom_baby |> 
  count(area, season)
desired_order <- c("North", "Mid", "South")
baby_area_count$area <- factor(baby_area_count$area, levels = desired_order)


ggplot(baby_area_count, aes(season, n, fill = area))+
  geom_col(position = "dodge")+
  scale_y_log10()+
  labs(
    title = "Whale mom and calf sighting in different areas",
    x = "Season",
    y = "Number of Sightings"
  )+
  theme_bw()
